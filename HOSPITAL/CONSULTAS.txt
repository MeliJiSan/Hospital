CONSULTAS:
-- Table: Hospital.consultas

-- DROP TABLE IF EXISTS "Hospital".consultas;

CREATE TABLE IF NOT EXISTS "Hospital".consultas
(
    id_consulta integer NOT NULL DEFAULT nextval('"Hospital".consultas_id_consultas_seq'::regclass),
    alergias text COLLATE pg_catalog."default" NOT NULL,
    observaciones_sintomas text COLLATE pg_catalog."default" NOT NULL,
    diagnostico text COLLATE pg_catalog."default" NOT NULL,
    fecha_registro timestamp with time zone NOT NULL,
    id_paciente integer NOT NULL,
    id_doctor integer NOT NULL,
    CONSTRAINT consultas_pkey PRIMARY KEY (id_consulta),
    CONSTRAINT id_doctor FOREIGN KEY (id_doctor)
        REFERENCES "Hospital".doctor (id_doctor) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
        NOT VALID,
    CONSTRAINT id_paciente FOREIGN KEY (id_paciente)
        REFERENCES "Hospital".pacientes (id_paciente) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
        NOT VALID
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS "Hospital".consultas
    OWNER to postgres;
